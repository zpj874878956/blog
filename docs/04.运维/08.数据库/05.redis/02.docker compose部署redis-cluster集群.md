# linux服务器arm架构快速安装redis cluster集群

docker-compose.yaml文件如下

```yaml
version: '3'

services:
  redis-1:
    image: redis:latest
    container_name: "redis-1"
    command: >
      redis-server
      --port 6379
      --cluster-enabled yes
      --cluster-config-file nodes.conf
      --cluster-node-timeout 5000
      --appendonly yes
      --requirepass test@123
      --masterauth test@456
    ports:
      - "6379:6379"
    volumes:
      - ./redis-1-data:/data
    networks:
      redis-net:
        ipv4_address: 172.23.0.2

  redis-2:
    image: redis:latest
    container_name: "redis-2"
    command: >
      redis-server
      --port 6379
      --cluster-enabled yes
      --cluster-config-file nodes.conf
      --cluster-node-timeout 5000
      --appendonly yes
      --requirepass test@123
      --masterauth test@456
    ports:
      - "6380:6379"
    volumes:
      - ./redis-2-data:/data
    networks:
      redis-net:
        ipv4_address: 172.23.0.3

  redis-3:
    image: redis:latest
    container_name: "redis-3"
    command: >
      redis-server
      --port 6379
      --cluster-enabled yes
      --cluster-config-file nodes.conf
      --cluster-node-timeout 5000
      --appendonly yes
      --requirepass test@123
      --masterauth test@456
    ports:
      - "6381:6379"
    volumes:
      - ./redis-3-data:/data
    networks:
      redis-net:
        ipv4_address: 172.23.0.4
networks:
  redis-net:
    driver: bridge
    ipam:
      driver: default
      config:
        - subnet: 172.23.0.0/16
```

创建集群：
由于现在有了密码，我们需要修改创建集群的命令：

```bash
docker exec -it redis-1 redis-cli -a test@123 --cluster create 172.23.0.2:6379 172.23.0.3:6379 172.23.0.4:6379 --cluster-replicas 0 --pass test@123
```

查看集群状态

```bash
docker exec -it $(docker ps -q -f name=redis-1) redis-cli -a test@123 cluster info
```
