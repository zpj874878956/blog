argocd部署

# 安装argo cd

# 创建ns

[root@k8s-master ~]# kubectl create namespace argocd
1

# 安装argocd

[root@k8s-master ~]# kubectl create -n argocd -f https://raw.githubusercontent.com/argoproj/argo-cd/stable/manifests/install.yaml
1

注意事项：默认下载的是最新版argocd，安装argocd时，务必参阅支持的k8s版本列表，否则会出现安装失败pod运行异常的情况。
参考文档：https://argo-cd.readthedocs.io/en/stable/operator-manual/installation/#supported-versions
由于k8s集群版本为1.23.16。因此安装的argo cd版本为2.4.28，yaml文件地址：https://github.com/argoproj/argo-cd/blob/v2.4.28/manifests/install.yaml

# web访问argocd

访问Argo server的方式有两种：

  • 通过web ui
  • 使用argocd 客户端工具

# 访问web ui(ingress方式)

访问web ui必须使用https方式访问，以traefik为例，创建ingressroute资源

# 创建证书文件
[root@k8s-master argo]# openssl req -x509 -nodes -days 365 -newkey rsa:2048 -keyout tls.key -out tls.crt -subj "/CN=argocd.test.com"
# 创建secret资源
[root@k8s-master argo]# kubectl create secret tls argocd-tls --cert=tls.crt --key=tls.key -n argocd
secret/myapp2-tls created
[root@k8s-master argo]# kubectl describe secrets argocd-tls -n argocd
Name:         argocd-tls
Namespace:    default
Labels:       <none>
Annotations:  <none>

Type:  kubernetes.io/tls

Data
====
tls.crt:  1131 bytes
tls.key:  1704 bytes
# 创建ingress资源
[root@k8s-master argo]# cat ingress.yaml 
apiVersion: traefik.containo.us/v1alpha1
kind: IngressRoute
metadata:
  name: argocd-tls
  namespace: argocd
spec:
  entryPoints:
    - websecure                    # 监听 websecure 这个入口点，也就是通过 443 端口来访问
  routes:
  - match: Host(`argocd.test.com`)
    kind: Rule
    services:
    - name: argocd-server
      port: 443
  tls:
    secretName: argocd-tls       # 指定tls证书名称
[root@k8s-master argo]# kubectl apply -f ingress.yaml 
ingressroute.traefik.containo.us/argocd-tls create

# 使用istio配置入口

需要几步来配置 HTTPS + 自签证书：

  1. 生成自签证书
# 生成私钥和自签证书
openssl req -x509 -nodes -days 365 -newkey rsa:2048 \
  -keyout argocd.key \
  -out argocd.crt \
  -subj "/CN=argocd.zzppjj.top/O=argocd"

# 创建 Kubernetes TLS Secret
kubectl create secret tls argocd-tls \
  --cert=argocd.crt \
  --key=argocd.key \
  -n argocd
1

  1. 更新 gateway.yaml
apiVersion: gateway.networking.k8s.io/v1
kind: Gateway
metadata:
  name: argocd-gateway
  namespace: argocd
spec:
  gatewayClassName: istio
  listeners:
  # HTTP 监听器 → 重定向到 HTTPS
  - name: http
    port: 80
    hostname: argocd.zzppjj.top
    protocol: HTTP
    allowedRoutes:
      namespaces:
        from: Same
  # HTTPS 监听器
  - name: https
    port: 443
    hostname: argocd.zzppjj.top
    protocol: HTTPS
    tls:
      mode: Terminate
      certificateRefs:
      - name: argocd-tls   # 对应上面创建的 Secret
        kind: Secret
    allowedRoutes:
      namespaces:
        from: Same
---
# HTTP → HTTPS 重定向
apiVersion: gateway.networking.k8s.io/v1
kind: HTTPRoute
metadata:
  name: argocd-http-redirect
  namespace: argocd
spec:
  parentRefs:
  - name: argocd-gateway
    namespace: argocd
    sectionName: http
  rules:
  - filters:
    - type: RequestRedirect
      requestRedirect:
        scheme: https
        statusCode: 301
---
# HTTPS 路由
apiVersion: gateway.networking.k8s.io/v1
kind: HTTPRoute
metadata:
  name: argocd-route
  namespace: argocd
spec:
  parentRefs:
  - name: argocd-gateway
    namespace: argocd
    sectionName: https
  rules:
  - matches:
    - path:
        type: PathPrefix
        value: /
    backendRefs:
    - name: argocd-server
      port: 80

3.ArgoCD Server 需要关闭内部 TLS

#ArgoCD Server 需要关闭内部 TLS
kubectl edit configmap argocd-cmd-params-cm -n argocd
添加 server.insecure: "true"：
apiVersion: v1
kind: ConfigMap
metadata:
  name: argocd-cmd-params-cm
  namespace: argocd
data:
  server.insecure: "true"    # ← 加这一行
或者用命令行直接 patch：
kubectl patch configmap argocd-cmd-params-cm -n argocd \
  --type merge \
  -p '{"data":{"server.insecure":"true"}}'
